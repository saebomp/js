AdjacentElementsProduct
Find largest adjacent product in array

const inputArray = [3,6,-2,-5,7,1]

const adjacentElementsProduct = inputArray => {

let largestProduct = inputArray[0] * inputArray[1]

//비교를 해서 더 큰 숫자를 찾기위해 넣음


for(let i = 0; i < inputArray.length; i++) { if(getProduct(inputArray, i) > largestProduct ) { largestProduct = getProduct(inputArray, i)}}

// 어레이를 돌려서 위의 숫자와 비교를 해준다음 더 크면 그 숫자를 반환함

return largestProduct

}

const getProduct = (inputArray, i) => ( inputArray[i] * inputArray[i + 1] )
