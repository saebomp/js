const result = document.getElementById('result')

const formatDate = (date = new Date()) => { //const date = new Date() 위에거 지우고 이렇게 써도됨 const days = date.getDate() const months = date.getMonth() + 1 const years = date.getFullYear() return `${days}/${months}/${years}`

}

const showDate = () => { result.innerHTML = `Today is ${formatDate()}` }
