Random number between two number
Random number between two number (이 예제는 주어진 두 값 사이의 난수를 생성한다. 함수의 반환값은 min보다 크거나 같으며, max보다 작다)

const result = document.getElementById('result')

const getRandomArbitrary = (min, max) => { return parseInt(Math.random() * (max - min) + min); // parseInt = change strings to number //최댓값은 제외, 최솟값은 포함

}

console.log(getRandomArbitrary(2,10))


const showResult = (a,b) => { result.innerHTML = getRandomArbitrary(a,b)

}

showResult(1,10)
