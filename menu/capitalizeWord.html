capitalze Words

1.

const capitalzeWords = (str) => { let words = str.split(" ").map(word => { return word.charAt(0).toUpperCase() + word.slice(1) }) return words.join(" ")

}

2.

const capitalzeWords = str => { let words = str.split(" ").map(word=>{ let firstLetter = word.slice(0, 1) let rest = word.slice(1) firstLetter = firstLetter.toUpperCase() return `${firstLetter}${rest}` }) return words.join(" ")

}

//두 개 결과 똑같음



console.log(capitalzeWords("I woke up early today"))

console.log(capitalzeWords("I went straight to the beach"))

charAt() 함수는 문자열에서 특정 인덱스에 위치하는 유니코드 단일문자를 반환합니다.

slice() 메서드는 어떤 배열의 begin부터 end까지(end 미포함)에 대한 얕은 복사본을 새로운 배열 객체로 반환합니다. 원본 배열은 바뀌지 않습니다.
